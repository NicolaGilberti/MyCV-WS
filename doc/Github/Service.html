def fetch(opts={})
  params = opts.merge(page: 0)
               .map { |k, v| "#{URI.encode(k.to_s)}=#{URI.encode(v.to_s)}" }
               .join('&')

  json = JSON.parse(open("#{Sources::GITHUB}/positions.json?#{params}").read)

  raw_jobs = json.map { |job| Job.new(job) }
  @jobs = adapt_github_json(raw_jobs)
end
